34  public class AppassemblerBooterTest

45      public void testSystemProperties()
46          throws Exception
47      {
48          System.setProperty( "app.name", "org/codehaus/mojo/appassembler/booter/app" );
49  
50          // When running from Maven this property will be set
51          if ( System.getProperty( "basedir" ) == null )
52          {
53              System.setProperty( "basedir", new File( "" ).getAbsolutePath() );
54          }
55  
56          AppassemblerBooter.setup();
57  
58          assertEquals( "System property bar is not set", "foo", System.getProperty( "bar" ) );
59  
60          assertEquals( "System property foo is not set", "bar", System.getProperty( "foo" ) );
61      }
62  
63      public void testRun()
64          throws Throwable
65      {
66          System.setProperty( "app.name", "org/codehaus/mojo/appassembler/booter/app" );
67  
68          URLClassLoader classLoader = AppassemblerBooter.setup();
69  
70          System.out.println( "classLoader = " + classLoader );
71  
72          try
73          {
74              AppassemblerBooter.executeMain( classLoader, new String[0] );
75          }
76          catch ( InvocationTargetException e )
77          {
78              throw e.getTargetException();
79          }
80  
81          Class klass = classLoader.loadClass( "org.codehaus.mojo.appassembler.booter.DummyMain" );
82  
83          System.out.println( "klass = " + klass );
84  
85          Field field = klass.getField( "kickAss" );
86          assertTrue( field.getBoolean( klass ) );
87      }
88  
89      public void testLargeConfig()
90          throws Exception
91      {
92          System.setProperty( "app.name", "org/codehaus/mojo/appassembler/booter/largeApp" );
93          System.setProperty( "app.booter.debug", "true" );
94          URLClassLoader classLoader = AppassemblerBooter.setup();
95          ArrayList<URL> urls = new ArrayList<URL>();
96          urls.addAll( Arrays.asList( classLoader.getURLs() ) );
97          for ( URL url : urls  )
98          {
99              if ( url.getPath().contains( "appassembler-booter/adgenerator-api-1.0-I11-SNAPSHOT.jar" ) )
100             {
101                 fail( "Broken path" );
102             }
103         }
104 
105     }

107     public void testMainWithArgs()
108         throws Throwable
109     {
110         System.setProperty( "app.name", "org/codehaus/mojo/appassembler/booter/appWithArgs" );
111 
112         URLClassLoader classLoader = AppassemblerBooter.setup();
113 
114         try
115         {
116             AppassemblerBooter.executeMain( classLoader, new String[] { "second argument" } );
117         }
118         catch ( InvocationTargetException e )
119         {
120             fail( "Unexpected failure: " + e );
121         }
122 
123         classLoader = AppassemblerBooter.setup();
124 
125         try
126         {
127             AppassemblerBooter.executeMain( classLoader, new String[0] );
128             fail( "Should have thrown exception" );
129         }
130         catch ( InvocationTargetException e )
131         {
132             assertTrue( "Unexpected exception " + e.getCause(),
133                         e.getCause().getMessage().startsWith( "Expected two arguments" ) );
134         }
135 
136     }
